## 表示項目

### 一覧カラム

| カラム | 表示名 | 型 | ソート | 幅 | 備考 |
|--------|--------|-----|--------|------|------|
| leaveType | 休暇種別 | text | - | 120px | 表示名マッピングで変換 |
| leavePeriod | 取得期間 | date | ✅ | 140px | 単日: MM/DD(曜)、期間: MM/DD〜MM/DD |
| timeSlot | 取得時間帯 | text | - | 110px | HOURLYのみ HH:mm〜HH:mm、他は "−" |
| submittedAt | 申請日 | date | ✅ (デフォルト降順) | 100px | MM/DD 形式 |
| approverId | 承認者 | text | - | 80px | 承認者名を表示 |
| status | ステータス | badge | - | 100px | 色分けバッジ |

### 休暇種別の表示名マッピング

| 値 | 表示名 |
|----|--------|
| ANNUAL | 年次有給 |
| HALF_DAY_AM | 午前半休 |
| HALF_DAY_PM | 午後半休 |
| HOURLY | 時間休 |
| SPECIAL_CONDOLENCE | 慶弔休暇 |
| SPECIAL_REFRESH | リフレッシュ休暇 |

### ステータスバッジ

| 値 | 表示 | 色 | 背景色 |
|----|------|-----|--------|
| SUBMITTED | 申請中 | blue-700 | blue-100 |
| APPROVED | 承認済 | green-700 | green-100 |
| REJECTED | 却下 | red-700 | red-100 |
| CANCELLED | 取消済 | gray-700 | gray-100 |

### 取得期間の表示ロジック

| 条件 | 表示形式 | 例 |
|------|---------|-----|
| from == to（単日） | MM/DD(曜) | 02/14(金) |
| from != to（期間） | MM/DD〜MM/DD | 02/03〜02/04 |

### 取得時間帯の表示ロジック

| 条件 | 表示形式 | 例 |
|------|---------|-----|
| leaveType == HOURLY | HH:mm〜HH:mm | 10:00〜12:00 |
| leaveType != HOURLY | "−"（ハイフン） | − |

## 状態管理

```typescript
/** 休暇種別 */
type LeaveType =
  | 'ANNUAL'
  | 'HALF_DAY_AM'
  | 'HALF_DAY_PM'
  | 'HOURLY'
  | 'SPECIAL_CONDOLENCE'
  | 'SPECIAL_REFRESH';

/** 休暇申請ステータス */
type LeaveRequestStatus =
  | 'SUBMITTED'
  | 'APPROVED'
  | 'REJECTED'
  | 'CANCELLED';

/** ソート方向 */
type SortOrder = 'asc' | 'desc';

/** ソート可能カラム */
type SortableColumn =
  | 'leavePeriod'
  | 'submittedAt';

/** 休暇申請一覧アイテム */
interface LeaveRequestItem {
  requestId: string;               // 休暇申請ID
  employeeId: string;              // 申請者ID
  employeeName: string;            // 申請者名（従業員マスタから導出）
  leaveType: LeaveType;
  leavePeriod: {
    from: string;                  // ISO 8601 date
    to: string;                    // ISO 8601 date
  };
  timeSlot: {
    startTime: string;             // HH:mm
    endTime: string;               // HH:mm
  } | null;                        // HOURLYのみ、他はnull
  status: LeaveRequestStatus;
  submittedAt: string;             // ISO 8601 datetime
  approverId: string | null;
  approverName: string | null;     // 従業員マスタから導出
}

/** 検索パラメータ */
interface SearchParams {
  dateFrom: string;                // ISO 8601 date、デフォルト: 当月1日
  dateTo: string;                  // ISO 8601 date、デフォルト: 当月末日
  leaveType: LeaveType | null;     // null = 全て
  status: LeaveRequestStatus | null; // null = 全て
}

/** ページネーション */
interface PaginationState {
  page: number;                    // 現在ページ (0-based)
  size: number;                    // 1ページあたり件数 (デフォルト: 20)
  sort: SortableColumn;            // ソートカラム (デフォルト: submittedAt)
  order: SortOrder;                // ソート方向 (デフォルト: desc)
}

/** 画面全体の状態 */
interface LeaveRequestListState {
  /** 検索条件 */
  searchParams: SearchParams;

  /** ページネーション */
  pagination: PaginationState;

  /** データ */
  data: {
    content: LeaveRequestItem[];
    page: {
      number: number;
      size: number;
      totalElements: number;
      totalPages: number;
    };
    isLoading: boolean;
    error: {
      message: string;
      retryAction: (() => void) | null;
    } | null;
  };
}
```
